# PowerShell script for downloading climate data# Blatchford station id :27214# Namao station id : 30907# Define the station ID and fixed parameters$stationID = 30907$timeframe = 2# Loop through years and monthsfor ($year = 2011; $year -le 2019; $year++) {    for ($month = 1; $month -le 1; $month++) {        # Construct the download URL        $url = "https://climate.weather.gc.ca/climate_data/bulk_data_e.html?format=csv&stationID=$stationID&Year=$year&Month=$month&Day=14&timeframe=$timeframe&submit=Download+Data"        # Use Invoke-WebRequest to download the data        Invoke-WebRequest -Uri $url -OutFile "weather_data_${year}_${month}.csv"    }}
